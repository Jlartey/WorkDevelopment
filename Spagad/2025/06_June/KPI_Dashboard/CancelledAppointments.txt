Sub generateReport()
  Dim sql, rst, cnt, periodStart, periodEnd, SpecialistTypeID
  Set rst = CreateObject("ADODB.Recordset")
  
  periodStart = Request.QueryString("printfilter")
  periodEnd = Request.QueryString("printfilter1")


  sql = "SELECT "
  sql = sql & "     AppointmentName, "
  sql = sql & "     convert(VARCHAR(20), AppointDate, 103) AppointmentDate, "
  sql = sql & "     SpecialistID, appointmentcatid, appointmentcattypeid, Remarks "
  sql = sql & " FROM Appointment"
  sql = sql & " WHERE AppointmentStatusID = 'A004' "
  sql = sql & " AND appointdate BETWEEN '" & periodStart & "' AND '" & periodEnd & "'"

  cnt = 0

  With rst
    .open qryPro.FltQry(sql), conn, 3, 4
      If .RecordCount > 0 Then
      
        response.write "<table class = 'myTable' > "
        response.write "    <thead> "
        response.write "    <tr class = 'myTable'>"
        response.write "        <th colspan = '13'>TOTAL APPOINTMENTS BETWEEN " & periodStart & " and " & periodEnd & "</th>"
        response.write "    </tr>"
        response.write "    <tr class = 'tHead'> "
        response.write "        <th>#</th> "
        response.write "        <th style=""text-align: left;"">Name</th> "
        response.write "        <th style=""text-align: left;"">Appointment Date</th> "
        response.write "        <th style=""text-align: left;"">Specialist</th> "
        response.write "        <th style=""text-align: left;"">Department</th> "
        response.write "        <th style=""text-align: left;"">Purpose</th> "
        response.write "    </tr> "
        response.write "    </thead><tbody> "

        .movefirst
        Do While Not .EOF
          cnt = cnt + 1

          response.write "  <tr class = 'queryData'> "
          response.write "      <td>" & (cnt) & "</td> "
          response.write "      <td style=""text-align: left;"">" & .fields("AppointmentName") & "</td> "
          response.write "      <td style=""text-align: left;"">" & .fields("AppointmentDate") & "</td> "
          response.write "      <td style=""text-align: left;"">" & GetComboName("Specialist", .fields("SpecialistID")) & "</td> "
'          response.write "      <td style=""text-align: left;"">" & GetComboName("appointmentcat", .fields("appointmentcatid")) & "</td> "
          response.write "      <td style=""text-align: left;"">" & GetComboName("appointmentcattype", .fields("appointmentcattypeid")) & "</td> "
          response.write "      <td style=""text-align: left;"">" & .fields("Remarks") & "</td> "
          response.write "  </tr> "
          response.flush
          .MoveNext

        Loop
      End If
      response.write "</tbody></table>"
    .Close
    Set rst = Nothing
  End With
End Sub
